<script>
// ================= CONFIG =================
const SCRIPT_URL = "https://script.google.com/macros/s/AKfycbzBDzPluVKvEViM3bHakiIF37GJ2j2KlRFSF5XWhaLvbRELGUVMxIlwpB2SeE-BEaG8/exec";

const scoreAssignments = [
  { name: 'Way Ground - Respiration', full: 10 },
  { name: 'Blood Donation Role-Play Assessment', full: 10 },
  { name: 'Heart - Oral Test', full: 10 },
  { name: 'Digestive Kitchen Lab', full: 10 },
  { name: 'Poster - Immune System Diseases', full: 10 },
  { name: 'Kahoot - Excretory system', full: 10 },
  { name: 'Final Exam : Multiple-choice', full: 20 },
  { name: 'Final Exam : Subjective', full: 20 }
];

let allStudents = [];
let currentStudentId = "";

// ================= LOGIN =================
function teacherLogin() {
  Swal.fire({
    title: 'รหัสผ่านครู',
    input: 'password',
    confirmButtonText: 'เข้าสู่ระบบ',
    preConfirm: p => {
      if (p !== TEACHER_PASSWORD) {
        Swal.showValidationMessage('รหัสผ่านไม่ถูกต้อง');
      }
    }
  }).then(r => {
    if (r.isConfirmed) {
      localStorage.setItem("teacherLoggedIn","true");
      enableTeacher();
      Swal.fire("สำเร็จ","เข้าสู่ระบบแล้ว","success");
    }
  });
}

function enableTeacher(){
  const btn = document.getElementById("fetchBtn");
  btn.disabled = false;
  btn.classList.remove("opacity-50");
}

// ================= FETCH =================
function fetchFromSheet(){
  if (localStorage.getItem("teacherLoggedIn") !== "true") {
    Swal.fire("เฉพาะครู","กรุณาเข้าสู่ระบบ","warning");
    return;
  }

  const cb = "cb"+Date.now();
  window[cb] = res => {
    delete window[cb];
    if(res.status==="success"){
      allStudents = res.data;
      localStorage.setItem("sheetData",JSON.stringify(allStudents));
      if(currentStudentId) displayStudent(currentStudentId);
      Swal.fire("สำเร็จ","โหลดข้อมูลแล้ว","success");
    }
  };

  const s = document.createElement("script");
  s.src = `${SCRIPT_URL}?callback=${cb}`;
  document.body.appendChild(s);
}

// ================= DISPLAY =================
function checkScore(){
  const id = studentId.value.trim();
  if(!id) return;
  currentStudentId = id;

  coverPage.classList.add("hidden");
  dashboardPage.classList.remove("hidden");

  const cache = localStorage.getItem("sheetData");
  if(cache){
    allStudents = JSON.parse(cache);
    displayStudent(id);
  }
}

function displayStudent(id){
  const st = allStudents.find(s=>String(s["เลขประจำตัว"])===id);
  if(!st){
    noDataMessage.classList.remove("hidden");
    return;
  }

  noDataMessage.classList.add("hidden");
  studentInfo.classList.remove("hidden");
  scoreSummary.classList.remove("hidden");
  scoreTable.classList.remove("hidden");

  studentName.textContent = st["ชื่อ-สกุล"];
  displayStudentId.textContent = id;
  displayRoom.textContent = st["ห้อง"];
  displayNumber.textContent = st["เลขที่"];

  let total = 0;
  scoreTableBody.innerHTML="";

  scoreAssignments.forEach(a=>{
    const sc = Number(st[a.name])||0;
    total += sc;
    scoreTableBody.innerHTML += `
      <tr>
        <td class="p-2">${a.name}</td>
        <td class="p-2 text-center">${a.full}</td>
        <td class="p-2 text-center font-bold">${sc}</td>
      </tr>`;
  });

  totalScoreCard.textContent = total;
  percentCard.textContent = total+"%";
  displayGrade.textContent = calcGrade(total);
}

function calcGrade(s){
  if(s>=80)return"4";
  if(s>=75)return"3.5";
  if(s>=70)return"3";
  if(s>=65)return"2.5";
  if(s>=60)return"2";
  if(s>=55)return"1.5";
  if(s>=50)return"1";
  return"0";
}

function goBack(){
  location.reload();
}

if(localStorage.getItem("teacherLoggedIn")==="true"){
  enableTeacher();
}
